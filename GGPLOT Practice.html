Delhi = tcpd %>% 
  filter(State_Name == 'Delhi' & Year >= 2004 & Year <=2019 & Poll_No == 0)

Delhi_turnout = Delhi %>% 
  filter(Position==1)%>%
  group_by(Year, Constituency_Name)%>%
  summarise(turnout=mean(Turnout_Percentage,na.rm=T)) %>% 
  mutate (State = "Delhi")
